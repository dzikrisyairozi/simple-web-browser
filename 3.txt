import java.awt.*;
import java.awt.event.*;
import javax.swing.*;
import javax.swing.event.*;
import java.net.*;
import java.io.*;

public class App extends JFrame {
    private JTextField addressField;
    private JEditorPane displayPane;

    public App() {
        super("Simple Web Browser");

        addressField = new JTextField(30);
        addressField.addActionListener(new ActionListener() {
            public void actionPerformed(ActionEvent e) {
                loadPage(addressField.getText());
            }
        });

        displayPane = new JEditorPane();
        displayPane.setEditable(false);
        displayPane.addHyperlinkListener(new HyperlinkListener() {
            public void hyperlinkUpdate(HyperlinkEvent e) {
                if (e.getEventType() == HyperlinkEvent.EventType.ACTIVATED) {
                    loadPage(e.getURL().toString());
                }
            }
        });

        JScrollPane scrollPane = new JScrollPane(displayPane);

        getContentPane().add(addressField, BorderLayout.NORTH);
        getContentPane().add(scrollPane, BorderLayout.CENTER);

        setSize(640, 480);
        setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        setVisible(true);
    }

    private void loadPage(String url) {
        try {
            displayPane.setPage(url);
            addressField.setText(url);
        } catch (IOException e) {
            System.err.println("Error loading page: " + e);
        }
    }

    public static void main(String[] args) {
        new App();
    }
}
